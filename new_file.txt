def insertion_sort(array_datos)
    for i in 1...(array_datos.length)  # Step 1
        j = i # Step 2
        while j > 0 # Step 3
            if array_datos[j-1] > array_datos[j] # Step 4
                temp = array_datos[j]
                array_datos[j] = array_datos[j-1]
                array_datos[j-1] = temp
            else
                break
            end
            j = j - 1 # Step 5
        end
    end
    return array_datos
end
